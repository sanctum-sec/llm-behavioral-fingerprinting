        - 法律访问密钥使用RSA-4096封装
        """
        # 生成对称密钥用于内容加密（模拟SM9）
        content_key = secrets.token_bytes(32)
        
        # 加密内容
        encrypted_content = self._aes_encrypt(data, content_key)
        
        # 使用RSA封装法律访问密钥
        encrypted_legal_key = self._rsa_encrypt(
            legal_access_key.encode(),
            self.rsa_key.public_key()
        )
        
        return {
            "encrypted_content": base64.b64encode(encrypted_content).decode(),
            "encrypted_content_key": base64.b64encode(content_key).decode(),
            "encrypted_legal_key": base64.b64encode(encrypted_legal_key).decode(),
            "encryption_scheme": "HYBRID_SM9_RSA4096",
            "timestamp": int(time.time())
        }
    
    def multi_party_decryption_trigger(self, encrypted_data: Dict, 
                                      parties: List[str]) -> Optional[bytes]:
        """
        多方协同触发解密机制
        需要法院、监察机构、技术审查团队共同授权
        """
        if len(parties) < 3:
            return None
        
        # 验证各方授权（模拟）
        authorized = all(self._verify_party_authorization(p) for p in parties)
        
        if authorized:
            # 执行解密
            content_key = base64.b64decode(encrypted_data["encrypted_content_key"])
            encrypted_content = base64.b64decode(encrypted_data["encrypted_content"])
            return self._aes_decrypt(encrypted_content, content_key)
        
        return None
    
    def _generate_rsa_key(self) -> rsa.RSAPrivateKey:
        """生成RSA-4096密钥对"""
        return rsa.generate_private_key(
            public_exponent=65537,
            key_size=4096
        )
    
    def _aes_encrypt(self, data: bytes, key: bytes) -> bytes:
        """AES加密（模拟SM9）"""
        # 实际实现应使用国密SM9
        iv = secrets.token_bytes(16)
        cipher = Cipher(algorithms.AES(key), modes.CBC(iv))
        encryptor = cipher.encryptor()
        
        # PKCS7填充
        pad_len = 16 - (len(data) % 16)
        padded_data = data + bytes([pad_len] * pad_len)
        
        encrypted = encryptor.update(padded_data) + encryptor.finalize()
        return iv + encrypted
    
    def _aes_decrypt(self, encrypted_data: bytes, key: bytes) -> bytes:
        """AES解密"""
        iv = encrypted_data[:16]
        ciphertext = encrypted_data[16:]
        
        cipher = Cipher(algorithms.AES(key), modes.CBC(iv))
        decryptor = cipher.decryptor()
        
        decrypted = decryptor.update(ciphertext) + decryptor.finalize()
        
        # 去除填充
        pad_len = decrypted[-1]
        return decrypted[:-pad_len]
    
    def _rsa_encrypt(self, data: bytes, public_key) -> bytes:
        """RSA加密"""
        return public_key.encrypt(
            data,
            padding.OAEP(
                mgf=padding.MGF1(algorithm=hashes.SHA256()),
                algorithm=hashes.SHA256(),
                label=None
            )
        )
    
    def _verify_party_authorization(self, party: str) -> bool:
        """验证参与方授权（模拟）"""
        valid_parties = {"court", "supervisory_agency", "tech_review_team"}
        return party in valid_parties

class OnionRoutingNetwork:
    """基于洋葱路由的抗审查P2P网络"""
    
    def __init__(self):
        self.nodes = {}
        self.entry_nodes = []
        self.emergency_gateways = []
        
    def add_node(self, node_id: str, node_info: Dict):
        """添加节点到网络"""
        self.nodes[node_id] = {
            **node_info,
            "traffic_shaped": False,
            "encryption_mode": "standard",
            "metadata_log": []
        }
        
        if node_info.get("is_entry_node", False):
            self.entry_nodes.append(node_id)
            self._apply_traffic_shaping(node_id)
        
        if node_info.get("is_emergency_gateway", False):
            self.emergency_gateways.append(node_id)
    
    def send_message(self, message: Dict, emergency: bool = False) -> str:
        """
        通过洋葱路由发送消息
        支持紧急状态字段
        """
        # 选择入口节点
        entry_node = self._select_entry_node()
        
        # 构建洋葱路由路径
        path = self._build_onion_path(emergency)
        
        # 添加协议头
        protocol_header = {
            "version": "1.0",
            "emergency_field": 1 if emergency else 0,
            "timestamp": int(time.time()),
            "path_length": len(path)
        }
        
        # 多层加密（洋葱路由）
        encrypted_message = self._onion_encrypt(message, path, protocol_header)
        
        # 记录元数据（不包括内容）
        for node_id in path:
            self._log_metadata(node_id, {
                "message_id": hashlib.sha256(json.dumps(message).encode()).hexdigest()[:16],
                "timestamp": protocol_header["timestamp"],
                "emergency": emergency,
                "next_hop": path.index(node_id) + 1 if path.index(node_id) < len(path) - 1 else None
            })
        
        return encrypted_message
    
    def _apply_traffic_shaping(self, node_id: str):
        """实施流量整形以符合网络安全法"""
        self.nodes[node_id]["traffic_shaped"] = True
        self.nodes[node_id]["max_bandwidth"] = "10Mbps"
        self.nodes[node_id]["rate_limit"] = "1000req/min"
    
    def _select_entry_node(self) -> str:
        """选择入口节点"""
        if not self.entry_nodes:
            raise ValueError("No entry nodes available")
        return secrets.choice(self.entry_nodes)
    
    def _build_onion_path(self, emergency: bool) -> List[str]:
        """构建洋葱路由路径"""
        # 选择3-5个中继节点
        relay_nodes = [n for n in self.nodes.keys() 
                      if not self.nodes[n].get("is_entry_node", False)
                      and not self.nodes[n].get("is_emergency_gateway", False)]
        
        num_hops = secrets.randbelow(3) + 3  # 3-5跳
        path = secrets.sample(relay_nodes, min(num_hops, len(relay_nodes)))
        
        # 如果紧急状态，添加安全网关
        if emergency and self.emergency_gateways:
            path.append(secrets.choice(self.emergency_gateways))
        
        return path
    
    def _onion_encrypt(self, message: Dict, path: List[str], header: Dict) -> str:
        """多层加密（洋葱路由）"""
        # 从出口节点开始反向加密
        encrypted = json.dumps(message)
        
        for node_id in reversed(path):
            node_key = self.nodes[node_id].get("encryption_key", secrets.token_bytes(32))
            
            # 根据威胁级别选择加密模式
            if self.nodes[node_id].get("high_threat_region", False):
                encrypted = self._chaos_encrypt(encrypted, node_key)
            else:
                encrypted = self._standard_encrypt(encrypted, node_key)
            
            # 添加下一跳信息
            layer = {
                "next_hop": node_id,
                "encrypted_payload": encrypted,
                "header": header
            }
            encrypted = json.dumps(layer)
        
        return encrypted
    
    def _standard_encrypt(self, data: str, key: bytes) -> str:
        """标准加密"""
        # 简化为base64编码，实际应使用强加密
        return base64.b64encode(data.encode()).decode()
    
    def _chaos_encrypt(self, data: str, key: bytes) -> str:
        """混沌加密（高威胁地区使用）"""
        # 添加额外的混淆层
        chaotic_data = data[::-1] + secrets.token_hex(16)
        return base64.b64encode(chaotic_data.encode()).decode()
    
    def _log_metadata(self, node_id: str, metadata: Dict):
        """记录节点元数据（用于威胁分析）"""
        if node_id in self.nodes:
            self.nodes[node_id]["metadata_log"].append({
                **metadata,
                "log_timestamp": int(time.time())
            })

class FederatedThreatDetection:
    """基于联邦学习的智能威胁检测系统"""
    
    def __init__(self):
        self.local_models = {}
        self.global_model = None
        self.threat_patterns = self._load_threat_patterns()
        
    def analyze_behavior(self, node_data: Dict, behavior_logs: List[Dict]) -> Dict:
        """
        分析节点行为，检测潜在威胁
        包括"疑似境外势力干预"特征检测
        """
        threat_score = 0
        detected_patterns = []
        
        # 分析行为模式
        for log in behavior_logs:
            pattern_match = self._match_threat_pattern(log)
            if pattern_match:
                threat_score += pattern_match["risk_level"]
                detected_patterns.append(pattern_match["pattern_name"])
        
        # 使用本地模型评估
        if node_data["node_id"] in self.local_models:
            model_score = self.local_models[node_data["node_id"]].predict(behavior_logs)
            threat_score += model_score
        
        # 检查境外干预特征
        foreign_intervention = self._detect_foreign_intervention(behavior_logs)
        if foreign_intervention:
            threat_score += 10
            detected_patterns.append("suspected_foreign_intervention")
        
        return {
            "node_id": node_data["node_id"],
            "threat_score": min(threat_score, 100),
            "detected_patterns": detected_patterns,
            "timestamp": int(time.time()),
            "recommendation": self._generate_recommendation(threat_score)
        }
    
    def update_model_federated(self, node_id: str, model_update: Dict, signature: str):
        """联邦学习模型更新（加密签名确保完整性）"""
        if self._verify_update_signature(model_update, signature):
            # 合并模型更新
            if node_id not in self.local_models:
                self.local_models[node_id] = {}
            
            self.local_models[node_id].update(model_update)
            
            # 定期聚合到全局模型
            if len(self.local_models) >= 5:  # 每5个节点更新聚合一次
                self._aggregate_global_model()
    
    def _load_threat_patterns(self) -> List[Dict]:
        """加载威胁模式库"""
        return [
            {
                "pattern_name": "rapid_data_exfiltration",
                "indicators": ["high_upload_rate", "unusual_timing"],
                "risk_level": 8
            },
            {
                "pattern_name": "covert_channel_usage",
                "indicators": ["stealth_protocols", "encrypted_metadata"],
                "risk_level": 9
            },
            {
                "pattern_name": "protocol_anomaly",
                "indicators": ["malformed_packets", "unusual_ports"],
                "risk_level": 7
            }
        ]
    
    def _match_threat_pattern(self, log: Dict) -> Optional[Dict]:
        """匹配威胁模式"""
        for pattern in self.threat_patterns:
            if all(indicator in str(log) for indicator in pattern["indicators"]):
                return pattern
        return None
    
    def _detect_foreign_intervention(self, logs: List[Dict]) -> bool:
        """检测境外干预特征"""
        foreign_indicators = [
            "foreign_ip_range",
            "unusual_geo_pattern",
            "known_apt_tactics",
            "cross_border_coordination"
        ]
        
        log_str = json.dumps(logs).lower()
        return any(indicator in log_str for indicator in foreign_indicators)
    
    def _verify_update_signature(self, update: Dict, signature: str) -> bool:
        """验证模型更新签名"""
        # 实际实现应使用数字签名验证
        return signature.startswith("SIGNED_")
    
    def _aggregate_global_model(self):
        """聚合全局模型"""
        # 联邦学习聚合算法（简化）
        if self.local_models:
            # 实际应使用FedAvg等算法
            self.global_model = {
                "aggregated_at": int(time.time()),
                "node_count": len(self.local_models),
                "model_version": "1.0"
            }
    
    def _generate_recommendation(self, threat_score: int) -> str:
        """生成处理建议"""
        if threat_score >= 80:
            return "立即隔离节点，启动深度调查"
        elif threat_score >= 60:
            return "限制节点权限，加强监控"
        elif threat_score >= 40:
            return "警告节点管理员，增加日志记录"
        else:
            return "正常监控"

class SystemHealthDashboard:
    """系统健康度仪表盘"""
    
    def __init__(self):
        self.metrics = {
            "system_uptime": 0,
            "active_nodes": 0,
            "submissions_today": 0,
            "threat_alerts": 0,
            "social_stability_index": 100.0
        }
        self.history = []
        
    def update_metrics(self, network_stats: Dict, threat_stats: Dict):
        """更新系统指标"""
        self.metrics.update({
            "system_uptime": network_stats.get("uptime", self.metrics["system_uptime"]),
            "active_nodes": network_stats.get("active_nodes", 0),
            "submissions_today": network_stats.get("daily_submissions", 0),
            "threat_alerts": threat_stats.get("active_alerts", 0),
            "timestamp": int(time.time())
        })
        
        # 计算社会稳定性指数
        self.metrics["social_stability_index"] = self._calculate_stability_index()
        
        # 记录历史
        self.history.append(self.metrics.copy())
        
        return self.metrics
    
    def _calculate_stability_index(self) -> float:
        """计算社会稳定性指数"""
        base_score = 100.0
        
        # 威胁警报影响
        threat_penalty = self.metrics["threat_alerts"] * 0.5
        base_score -= min(threat_penalty, 20.0)
        
        # 系统可用性影响
        if self.metrics["active_nodes"] < 10:
            base_score -= 10.0
        
        # 提交量影响（适中为好）
        submissions = self.metrics["submissions_today"]
        if submissions > 1000:
            base_score -= (submissions - 1000) * 0.01
        elif submissions < 10:
            base_score -= 5.0
        
        return max(base_score, 0.0)
    
    def generate_report(self) -> Dict:
        """生成健康度报告"""
        return {
            "dashboard_version": "1.0",
            "current_metrics": self.metrics,
            "trends": self._calculate_trends(),
            "recommendations": self._generate_recommendations(),
            "last_updated": int(time.time())
        }
    
    def _calculate_trends(self) -> Dict:
        """计算趋势"""
        if len(self.history) < 2:
            return {"insufficient_data": True}
        
        recent = self.history[-1]
        previous = self.history[-2] if len(self.history) >= 2 else recent
        
        trends = {}
        for key in recent:
            if isinstance(recent[key], (int, float)) and isinstance(previous[key], (int, float)):
                change = recent[key] - previous[key]
                trends[f"{key}_trend"] = {
                    "change": change,
                    "percent_change": (change / previous[key] * 100) if previous[key] != 0 else 0
                }
        
        return trends
    
    def _generate_recommendations(self) -> List[str]:
        """生成系统优化建议"""
        recommendations = []
        
        if self.metrics["social_stability_index"] < 80:
            recommendations.append("社会稳定性指数较低，建议增加节点容量")
        
        if self.metrics["threat_alerts"] > 5:
            recommendations.append("威胁警报较多，建议加强安全监控")
        
        if self.metrics["active_nodes"] < 20:
            recommendations.append("活跃节点不足，建议扩展网络规模")
        
        if not recommendations:
            recommendations.append("系统运行正常，继续保持当前配置")
        
        return recommendations
